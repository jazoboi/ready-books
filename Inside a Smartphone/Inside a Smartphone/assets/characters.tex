% ------------------------------------------------------------
%  Shared helper: coloured circle with icon label
%  #1 = fill colour   #2 = icon (fontawesome or letter)
%  #3 = name label below
% ------------------------------------------------------------
\newcommand{\charicon}[3]{%
  \begin{tikzpicture}
    \fill[#1!80] (0,0) circle (0.9cm);
    \draw[white,line width=1.5pt] (0,0) circle (0.9cm);
    \node[font=\bfseries\large\color{white}] at (0,0) {#2};
    \node[font=\small\bfseries\color{DarkText},below] at (0,-1.05) {#3};
  \end{tikzpicture}%
}

% ============================================================
%  Individual character commands
% ============================================================

\newcommand{\charDot}{\charicon{SkyBlue}{\faDotCircle}{Dot}}

% Chip — the processor
\newcommand{\charChip}{\charicon{TechPurple}{\faMicrochip}{Chip}}

% Memo — memory and storage

\newcommand{\charLens}{\charicon{WarmOrange}{\faCamera}{Lens}}

% Zap — the battery
\newcommand{\charZap}{\charicon{SunYellow!80!WarmOrange}{\faBolt}{Zap}}

% Sensa — the sensors
\newcommand{\charSensa}{\charicon{LeafGreen}{\faSlidersH}{Sensa}}

% Appy — apps
\newcommand{\charAppy}{\charicon{CoralRed}{\faMobile}{Appy}}

\newcommand{\charOrbi}{\charicon{SmartBlue}{\faSatellite}{Orbi}}

% ============================================================
%  \characterparade — all nine characters in a single row
%  Used on the Meet Your Guides page.
% ============================================================
\newcommand{\characterparade}{%
  \begin{center}
    \begin{tikzpicture}
      % Row 1: Dot, Chip, Memo, Tappy, Lens
      \node at (0*2.4,0)   {\charDot};
      \node at (1*2.4,0)   {\charChip};
      \node at (2*2.4,0)   {\charMemo};
      \node at (3*2.4,0)   {\charTappy};
      \node at (4*2.4,0)   {\charLens};
      % Row 2: Zap, Sensa, Appy, Orbi (centred below)
      \node at (0.5*2.4,-2.8) {\charZap};
      \node at (1.5*2.4,-2.8) {\charSensa};
      \node at (2.5*2.4,-2.8) {\charAppy};
      \node at (3.5*2.4,-2.8) {\charOrbi};
    \end{tikzpicture}
  \end{center}
  \vspace{8pt}
}
