% ============================================================
%  assets/characters.tex — Robot World
%  TikZ character illustration commands.
%
%  Each character is drawn as a simple, friendly icon.
%  \characterparade  — draws all characters in a row on the
%                        Meet Your Guides page.
% ============================================================

% ------------------------------------------------------------
%  Shared helper: coloured circle with icon label
%  #1 = fill colour   #2 = icon (fontawesome or letter)
%  #3 = name label below
% ------------------------------------------------------------
\newcommand{\charicon}[3]{%
  \begin{tikzpicture}
    \fill[#1!80] (0,0) circle (0.9cm);
    \draw[white,line width=1.5pt] (0,0) circle (0.9cm);
    \node[font=\bfseries\large\color{white}] at (0,0) {#2};
    \node[font=\small\bfseries\color{DarkText},below] at (0,-1.05) {#3};
  \end{tikzpicture}%
}

% ============================================================
%  Individual character commands
% ============================================================

% Dot — glowing information packet (returning from Books 1 & 2)
\newcommand{\charDot}{\charicon{SkyBlue}{\faDotCircle}{Dot}}

% Bolt — factory robot arm
\newcommand{\charBolt}{\charicon{RoboGreen}{\faCog}{Bolt}}

% Whirr — drone
\newcommand{\charWhirr}{\charicon{WarmOrange}{\faHelicopter}{Whirr}}

% Sprocket — wheeled rover
\newcommand{\charSprocket}{\charicon{SunYellow!80!WarmOrange}{\faCarSide}{Sprocket}}

% Cora — humanoid companion
\newcommand{\charCora}{\charicon{SoftPurple}{\faUser}{Cora}}

% Sparks — micro-bot surgeon
\newcommand{\charSparks}{\charicon{CoralRed}{\faProcedures}{Sparks}}

% Tread — rescue robot
\newcommand{\charTread}{\charicon{SteelBlue}{\faBolt}{Tread}}

% Echo — home assistant robot
\newcommand{\charEcho}{\charicon{MetalGray}{\faVolumeUp}{Echo}}

% ============================================================
%  \characterparade — all eight characters in rows
%  Used on the Meet Your Guides page.
% ============================================================
\newcommand{\characterparade}{%
  \begin{center}
    \begin{tikzpicture}
      % Row 1: Dot, Bolt, Whirr, Sprocket
      \node at (0*2.4,0)   {\charDot};
      \node at (1*2.4,0)   {\charBolt};
      \node at (2*2.4,0)   {\charWhirr};
      \node at (3*2.4,0)   {\charSprocket};
      % Row 2: Cora, Sparks, Tread, Echo
      \node at (0.5*2.4,-2.8) {\charCora};
      \node at (1.5*2.4,-2.8) {\charSparks};
      \node at (2.5*2.4,-2.8) {\charTread};
      \node at (3.5*2.4,-2.8) {\charEcho};
    \end{tikzpicture}
  \end{center}
  \vspace{8pt}
}
